package org.fixsuite.message.parsers.fpl;

import java.util.Iterator;
import java.util.List;

import org.fixsuite.message.info.DictionaryInfo;
import org.fixsuite.message.info.MessageInfo;
import org.jdom.Element;

/**
 * Parses the MsgTypeParser.xml.
 * 
 * @author jramoyo
 */
public class MsgTypeParser extends AbstractParser {

    private static final String MSG_TYPE_ELEMENT = "MsgType";

    private static final String MESSAGE_NAME_ELEMENT = "MessageName";

    private static final String COMPONENT_TYPE_ELEMENT = "ComponentType";

    private static final String CATEGORY_ELEMENT = "Category";

    private static final String MSG_ID_ELEMENT = "MsgID";

    private static final String SECTION_ELEMENT = "Section";

    private static final String ABBR_NAME_ELEMENT = "AbbrName";

    private static final String NOT_REQ_XML_ELEMENT = "NotReqXML";

    private static final String OVERRIDE_ABBR_ELEMENT = "OverrideAbbr";

    private static final String VOLUME_ELEMENT = "Volume";

    /**
     * @see org.fixsuite.message.parsers.fpl.AbstractParser#load(org.fixsuite.message.info.DictionaryInfo,
     *      org.jdom.Element)
     */
    @SuppressWarnings("unchecked")
    protected void load(DictionaryInfo dictionary, Element root) {
        List nodes = root.getChildren();
        Element element;
        String messageType;
        String name;
        String componentType;
        String category;
        int id;
        String section;
        String abbreviation;
        boolean isNotRequiredXml;
        String overrideAbbreviation;
        String volume;
        MessageInfo message;
        Iterator i = nodes.iterator();
        while (i.hasNext()) {
            // Extract the values
            element = (Element) i.next();
            messageType = element.getChildText(MSG_TYPE_ELEMENT);
            name = element.getChildText(MESSAGE_NAME_ELEMENT);
            componentType = element.getChildText(COMPONENT_TYPE_ELEMENT);
            category = element.getChildText(CATEGORY_ELEMENT);
            id = Integer.parseInt(element.getChildText(MSG_ID_ELEMENT));
            section = element.getChildText(SECTION_ELEMENT);
            abbreviation = element.getChildText(ABBR_NAME_ELEMENT);
            if (element.getChildText(NOT_REQ_XML_ELEMENT) != null
                    && element.getChildText(NOT_REQ_XML_ELEMENT).equals("1")) {
                isNotRequiredXml = true;
            } else {
                isNotRequiredXml = false;
            }
            overrideAbbreviation = element.getChildText(OVERRIDE_ABBR_ELEMENT);
            volume = element.getChildText(VOLUME_ELEMENT);
            // "35 = n" indicates a non-FIX XML message
            if (!messageType.equals("n")) {
                // Insert the values
                message = new MessageInfo();
                message.setMessageType(messageType);
                message.setName(name);
                message.setComponentType(componentType);
                message.setCategory(category);
                message.setId(id);
                message.setSection(section);
                message.setAbbreviation(abbreviation);
                message.setNotRequiredXml(isNotRequiredXml);
                message.setOverrideAbbreviation(overrideAbbreviation);
                message.setVolume(volume);
                // Add to dictionary
                dictionary.addMessage(message);
            }
        }
        dictionary.incrementLoadCount();
    }

}
