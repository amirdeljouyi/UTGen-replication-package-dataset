package com.browsersoft.openhre.hl7.api.config;

/**
 * something wrong in config data reading process
 */
public class InvalidConfigDataStructureException extends Exception {

	private String postfix;
	private int messageCode;

    /**
     * construct exception
     * @param postfix
     * @param messageCode
     */
    public InvalidConfigDataStructureException( String postfix, int messageCode ) {
		this.postfix = postfix;
		this.messageCode = messageCode;
	}

    /**
     * return string value after message
     * @return value
     */
    public String getPostfix() {
		return postfix;
	}

    /**
     * return message code
     * @return message code
     */
    public int getMessageCode() {
		return messageCode;
	}

}
