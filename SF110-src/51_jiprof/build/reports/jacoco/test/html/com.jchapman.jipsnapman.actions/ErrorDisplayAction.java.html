package com.jchapman.jipsnapman.actions;

import java.text.MessageFormat;

import java.awt.event.ActionEvent;
import javax.swing.JOptionPane;

import org.yasl.arch.application.YASLSwingApplication;
import org.yasl.arch.impl.action.YASLSwingAbstractAction;
import org.yasl.arch.impl.event.YASLActionEvent;
import org.yasl.arch.resources.YASLResourceManager;
import com.jchapman.jipsnapman.utils.JIPSnapManConstants;

/**
 *
 * @author not attributable
 * @version 1.0
 */
public class ErrorDisplayAction extends YASLSwingAbstractAction {

    public ErrorDisplayAction(YASLSwingApplication yaslSwingApplication) {
        super("error display", true, yaslSwingApplication);
    }

    protected void performAction(ActionEvent actionEvent)
            throws Exception {
        YASLActionEvent yaslActionEvent = (YASLActionEvent)actionEvent;
        YASLSwingApplication swingApp = getYASLSwingApplication();
        YASLResourceManager resMan = swingApp.getResourceManager();
        String dlgTitleKey = (String)yaslActionEvent.getProperty(JIPSnapManConstants.KEY_ERR_DLGTITLE);
        String mssgKey = (String)yaslActionEvent.getProperty(JIPSnapManConstants.KEY_ERR_MSSG);
        Object[] mssgValues = (Object[])yaslActionEvent.getProperty(JIPSnapManConstants.KEY_ERR_VALUES);
        String message;
        String defBundle = resMan.getDefaultBundle();
        String dlgTitle = resMan.getString(defBundle, dlgTitleKey);
        if(mssgValues != null) {
            message = MessageFormat.format(resMan.getString(defBundle, mssgKey), mssgValues);
        }
        else {
            message = resMan.getString(defBundle, mssgKey);
        }
        JOptionPane.showMessageDialog(swingApp.getJFrame(),
                                      message,
                                      dlgTitle,
                                      JOptionPane.ERROR_MESSAGE);
    }
}
