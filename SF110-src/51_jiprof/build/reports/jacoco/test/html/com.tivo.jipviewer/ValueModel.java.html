package com.tivo.jipviewer;

import java.util.ArrayList;
import java.util.List;

class ValueModel<Type> {
    Type mValue;
    List<ChangeListener> mvListener = new ArrayList<ChangeListener>();

    public void setValue(Type value) {
        if (mValue != value) {
            mValue = value;
            dispatchChangeEvent();
        }
    }

    public Type getValue() {
        return mValue;
    }

    public void addChangeListener(ChangeListener listener) {
        mvListener.add(listener);
    }

    public void removeChangeListener(ChangeListener listener) {
        mvListener.remove(listener);
    }

    private void dispatchChangeEvent() {
        for (ChangeListener listener : mvListener) {
            listener.changed(this);
        }
    }
};
