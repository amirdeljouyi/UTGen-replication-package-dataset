package de.paragon.explorer.model;

import java.lang.reflect.Array;

public class ArrayAttributeModel extends AttributeModel {
	private int	pos;

	/**
	 * ArrayAttributeModel constructor comment.
	 */
	public ArrayAttributeModel() {
		super();
	}

	private int getPos() {
		return this.pos;
	}

	@Override
	public Object getValue() {
		Object value;
		value = Array.get(this.getObjectModel().getObject(), this.getPos());
		if (value == null) {
			value = NullObject.getNullObject();
		}
		return value;
	}

	public void setPos(int newPos) {
		this.pos = newPos;
	}
}
