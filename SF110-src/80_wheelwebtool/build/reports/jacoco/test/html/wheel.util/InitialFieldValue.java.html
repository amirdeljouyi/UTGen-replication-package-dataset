package wheel.util;

import java.io.Serializable;

public class InitialFieldValue implements Serializable {
    private Object emptyValue;
    private String stringReplacement;

    public InitialFieldValue(Object emptyValue, String stringReplacement) {
        this.emptyValue = emptyValue;
        this.stringReplacement = stringReplacement;
    }

    public InitialFieldValue(String stringReplacement) {
        this.stringReplacement = stringReplacement;
    }

    public Object getEmptyValue() {
        return emptyValue;
    }

    public String getStringReplacement() {
        return stringReplacement;
    }
}
