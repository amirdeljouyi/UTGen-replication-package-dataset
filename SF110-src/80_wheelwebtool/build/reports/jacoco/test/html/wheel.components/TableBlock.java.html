package wheel.components;

/**
 * A special case of a Block component that is used internally inside a Table-component. The reason for this is to control the use of tags
 * that can be used only inside a Table.
 *
 * @author Henri Frilund
 */
public class TableBlock extends Block {

    TableBlock(Component parent) {
        super(parent);
    }

    TableBlock(Component parent, String componentId) {
        super(parent, componentId);
    }

    public TableRow tr() {
        TableRow tableRow = new TableRow(this);
        add(tableRow);
        return tableRow;
    }

    public Any col() {
        Any col = create().wAny("col");
        add(col);
        return col;
    }


    @Override
    public TableBlock id(String componentId) {
        return (TableBlock)super.id(componentId);
    }


    @Override
    public TableBlock renderHint(CharSequence renderHint) {
        return (TableBlock)super.renderHint(renderHint);
    }
}
