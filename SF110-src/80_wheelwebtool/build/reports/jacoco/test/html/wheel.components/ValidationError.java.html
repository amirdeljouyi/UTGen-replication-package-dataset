package wheel.components;

/**
 * Represents a validation error on a FormElement class. Many validation errors can be attached to single FormElement.
 *
 * @author Henri Frilund
 */
public class ValidationError {
    private String message;
    private Component source;

    public ValidationError(String message, Component source) {
        this.message = message;
        this.source = source;
    }

    /**
     * Returns the validation message (set by calling FormElement#addValidationRule(String, String)) without
     * evaluating it.
     * @return
     */
    public String getMessage() {
        return message;
    }

    /**
     * Returns the component that caused this error.
     * @return
     */
    public Component getSource() {
        return source;
    }
}
