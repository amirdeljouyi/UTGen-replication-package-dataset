package net.kencochrane.a4j.beans;

import java.io.Serializable;
import java.util.ArrayList;

/**
 * http://www.KenCochrane.net
 * Ken Cochrane
 * Date: May 24, 2003
 * Time: 6:22:26 PM
 *
 *
 */
public class FullProduct implements Serializable {
    ProductDetails details = new ProductDetails();
    ArrayList accessories = new ArrayList();
    ArrayList similarItems = new ArrayList();

    public ProductDetails getDetails() {
        return details;
    }

    public void setDetails(ProductDetails details) {
        this.details = details;
    }

    public ArrayList getAccessories() {
        return accessories;
    }

    public void setAccessories(ArrayList accessories) {
        this.accessories = accessories;
    }

    public ArrayList getSimilarItems() {
        return similarItems;
    }

    public void setSimilarItems(ArrayList similarItems) {
        this.similarItems = similarItems;
    }

    public void addAccessory(MiniProduct product) {
        accessories.add(product);
    }

    public void addSimilarItem(MiniProduct item) {
        similarItems.add(item);
    }

    public void printFullProduct() {
        System.out.println(details + "\n");

        System.out.println("-- Accessories --");
        if (accessories != null && accessories.size() > 0) {
            for (int x = 0; x < accessories.size(); x++) {
                System.out.println(accessories.get(x) + "\n");
            }
        }

        System.out.println("-- Similar Products --");
        if (similarItems != null && similarItems.size() > 0) {
            for (int x = 0; x < similarItems.size(); x++) {
                System.out.println(similarItems.get(x) + "\n");
            }
        }
    }
}
