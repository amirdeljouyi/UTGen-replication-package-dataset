package net.sf.sugar.fspath;

/**
 *  An enum of the attribute names to specify for a specific file.
 *  These are just taken from java.io.File
 *
 *  @author keith
 */
public enum FSPathAttributes {

    canRead(),
    canWrite(),
    exists(),
    absolutePath(),
    canonicalPath(),
    name(),
    parent(),
    path(),
    isAbsolute(),
    isDirectory(),
    isFile(),
    isHidden(),
    lastModified(),
    length(),
    file(), //name of a file tag
    dir();


}
