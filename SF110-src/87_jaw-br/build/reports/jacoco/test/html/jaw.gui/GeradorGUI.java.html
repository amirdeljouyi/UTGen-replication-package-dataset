    private void ajustarLargura() {
        TableColumn col = jTable1.getColumnModel().getColumn(0);
        col.setPreferredWidth(5);
        col = jTable2.getColumnModel().getColumn(0);
        col.setPreferredWidth(5);
    }

    private void listarModelos() {
        ((javax.swing.table.DefaultTableModel) jTable1.getModel()).setRowCount(0);
        for (ModeloDeAplicacao modelo: modelos) {
            ((javax.swing.table.DefaultTableModel) jTable1.getModel()).addRow(new Object[]{new Boolean(false), modelo.getNome(), modelo.getDescricao()});
        }
    }

    private void listarEntidades() {
        ((javax.swing.table.DefaultTableModel) jTable2.getModel()).setRowCount(0);
        for (int i = 0; i < entidades.size(); i++) {
            ((javax.swing.table.DefaultTableModel) jTable2.getModel()).addRow(new Object[]{new Boolean(false), entidades.get(i)});
        }
    }

    private void jButton1ActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_jButton1ActionPerformed
        // Gerar
        List<Entidade> selecionadas = new Vector();

        for (int i = 0; i < jTable2.getRowCount(); i++) {
            if (!((Boolean) jTable2.getModel().getValueAt(i, 0)).booleanValue()) {
                continue;
            }
            selecionadas.add((Entidade) jTable2.getModel().getValueAt(i, 1));
        }

        for (int i = 0; i < jTable1.getRowCount(); i++) {
            if (!((Boolean) jTable1.getModel().getValueAt(i, 0)).booleanValue()) {
                continue;
            }
            ModeloDeAplicacao modelo = modelos.get(i);
            new jaw.saida.Gerador(selecionadas, modelo);
        }
    }//GEN-LAST:event_jButton1ActionPerformed

    private void alterarModeloActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_alterarModeloActionPerformed
        // alterar template
        for (int i = 0; i < jTable1.getRowCount(); i++) {
            if (!((Boolean) jTable1.getModel().getValueAt(i, 0)).booleanValue()) {
                continue;
            }
            ModeloDeAplicacao modelo = modelos.get(i);
            new CriarModeloDeAplicacao(modelo).setVisible(true);
        }
}//GEN-LAST:event_alterarModeloActionPerformed

    private void jButton4ActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_jButton4ActionPerformed
        // inverter selecao - templates
        for (int i = 0; i < jTable1.getRowCount(); i++) {
            jTable1.getModel().setValueAt(new Boolean(!((Boolean) jTable1.getModel().getValueAt(i, 0)).booleanValue()), i, 0);

        }
    }//GEN-LAST:event_jButton4ActionPerformed
