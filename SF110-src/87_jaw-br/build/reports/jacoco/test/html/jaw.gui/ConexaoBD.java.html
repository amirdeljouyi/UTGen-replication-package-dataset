    private void driverFocusLost(java.awt.event.FocusEvent evt) {//GEN-FIRST:event_driverFocusLost
        try {
            if (this.driver.getModel().getSelectedItem().toString().equals("org.postgresql.Driver")) {
                this.caminho.setText("jdbc:postgresql://localhost:5432/");
            } else if (this.driver.getModel().getSelectedItem().toString().equals("com.mysql.jdbc.Driver")) {
                this.caminho.setText("jdbc:mysql://localhost/");
            }
        } catch (Exception e) {
            jaw.Main.janelaPrincipal.log(e.toString());
        }
}//GEN-LAST:event_driverFocusLost

    private void conectarActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_conectarActionPerformed
        // TODO add your handling code here:
        if (!this.driver.getSelectedItem().toString().equals("Selecione o Driver")) {
            jaw.entrada.Conexao conexao = new jaw.entrada.Conexao(this.driver.getSelectedItem().toString(),
                    this.caminho.getText(), this.base.getText(),
                    this.usuario.getText(), this.senha.getText());

            conexao.conectar();
            conectado = true;
            this.setVisible(false);
            this.processarEntidade.setNomeArquivoSalvo("");
            while (conexao.isConectado()) {
                
                this.processarEntidade.carregarEntidades(conexao.listarEntidades());
            }
            if (conexao.getAlteracao()) {
                JOptionPane.showMessageDialog(this, "A conexão detectou tipos desconhecidos pela ferramenta.\n" +
                        "É recomendável ajustar os mapeamentos e reimpotar os dados.\n" +
                        "Clique em Configurações -> Mapeamento para ajustar os mapeamentos.");
            }
        } else {
            JOptionPane.showMessageDialog(null, "Por Favor Selecione Um Driver!");
            return;
        }         
}//GEN-LAST:event_conectarActionPerformed

    private void conexoesActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_conexoesActionPerformed
        //Conexoes
        if (this.conexoes.getSelectedIndex() < 0) {
            return;
        }
        jaw.configuracoes.ConexaoBD con = lista.get(this.conexoes.getSelectedIndex());
        caminho.setText(con.getUrl());
        usuario.setText(con.getUsuario());
        senha.setText(con.getSenha());
        driver.getModel().setSelectedItem(con.getDriver());
        base.setText(con.getBase());
}//GEN-LAST:event_conexoesActionPerformed

    private void carregarConexoes() {
        lista = jaw.configuracoes.ConexaoBD.listar();
        this.conexoes.removeAllItems();
        for (jaw.configuracoes.ConexaoBD con : lista) {
            this.conexoes.addItem(con);
        }
    }

    private void jButton1ActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_jButton1ActionPerformed
        //Salvar
        String nome = JOptionPane.showInputDialog(this, "Entre com um nome para esta conexão");
        jaw.configuracoes.ConexaoBD con = new jaw.configuracoes.ConexaoBD();
        con.setUrl(caminho.getText());
        con.setUsuario(usuario.getText());
        con.setSenha(senha.getText());
        con.setDriver(driver.getModel().getSelectedItem().toString());
        con.setBase(base.getText());
        con.setNome(nome);
        con.adicionar();
        this.carregarConexoes();
    }//GEN-LAST:event_jButton1ActionPerformed

    private void jButton2ActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_jButton2ActionPerformed
        //Remover
        lista.get(this.conexoes.getSelectedIndex()).remover();
        this.carregarConexoes();
    }//GEN-LAST:event_jButton2ActionPerformed
