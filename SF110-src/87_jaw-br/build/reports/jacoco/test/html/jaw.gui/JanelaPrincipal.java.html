    private void carregarArquivosRecentes() {
        menuRecentes.removeAll();
        List<String> arquivos = ArquivoRecente.getArquivos();
        for (int i = 0; i < arquivos.size(); i++) {
            JMenuItem menuItem = new JMenuItem(arquivos.get(i));
            menuItem.addActionListener(new java.awt.event.ActionListener() {

                public void actionPerformed(java.awt.event.ActionEvent evt) {
                    abrirRecente(evt);
                }
            });
            menuRecentes.add(menuItem);
        }
        menuRecentes.add(jMenuItem5);
    }

    private void abrirRecente(java.awt.event.ActionEvent evt) {
        ProcessarEntidadesTabela processar = new ProcessarEntidadesTabela();
        int pos = jTabbedPane1.getTabCount();
        jTabbedPane1.add(processar, pos);
        new Abrir(processar, ((JMenuItem) evt.getSource()).getText());
        jTabbedPane1.setTitleAt(pos, processar.getNomeArquivoSalvo());
        jTabbedPane1.setSelectedIndex(pos);
    }

    public void log(String texto) {
        this.jTextArea1.setText(jTextArea1.getText() + "\n" + texto);
    }

    private void abrir(int opcao) {
        ProcessarEntidadesTabela processar = new ProcessarEntidadesTabela();
        int pos = jTabbedPane1.getTabCount();
        jTabbedPane1.add(processar, pos);
        new Abrir(processar, opcao);
        jTabbedPane1.setTitleAt(pos, processar.getNomeArquivoSalvo());
        jTabbedPane1.setSelectedIndex(pos);
        ArquivoRecente.adicionarArquivo(processar.getNomeArquivoSalvo());
        carregarArquivosRecentes();
    }

    private void sairMousePressed(java.awt.event.MouseEvent evt) {//GEN-FIRST:event_sairMousePressed
        int opc = javax.swing.JOptionPane.showConfirmDialog(null, "Realmente deseja encerrar a Jaw?", "Jaw - Sair", JOptionPane.YES_NO_OPTION);
        if (opc == 0 && jTabbedPane1.getTabCount() > 0) {//JOptionPane.YES_NO_OPTION) {
            if (((ProcessarEntidadesTabela) jTabbedPane1.getSelectedComponent()).getNomeArquivoSalvo().equals("") && ((ProcessarEntidadesTabela) jTabbedPane1.getSelectedComponent()).getEntidades().size() == 0) {
                System.exit(0);
            } else {

                    new jaw.entrada.Salvar(((ProcessarEntidadesTabela) jTabbedPane1.getSelectedComponent()), 30);
                    System.exit(0);
                } else {

                }
            }
        } else {
            return;
        }

    private void jMenuItem2ActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_jMenuItem2ActionPerformed
        // Visualizar Tabela
        ProcessarEntidades processar = new ProcessarEntidadesTabela();
        List<Entidade> entidades = new Vector();
        int selecionado = jTabbedPane1.getSelectedIndex();
        if (selecionado < 0) {
            return;
        }
        try {
            entidades = ((ProcessarEntidades) jTabbedPane1.getTabComponentAt(selecionado)).getEntidades();
        } catch (Exception e) {
            jaw.Main.janelaPrincipal.log(e.toString());
            entidades = new Vector();
        }

        processar.setEntidades(entidades);
        int pos = jTabbedPane1.getTabCount();
        jTabbedPane1.add(processar, pos);
        jTabbedPane1.setTitleAt(pos, processar.getNomeArquivoSalvo());
        jTabbedPane1.setSelectedIndex(pos);   
    }//GEN-LAST:event_jMenuItem2ActionPerformed

    private void jButton3ActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_jButton3ActionPerformed
        try {
            if (getFormAtual().getEntidadeSelecionada() < 0) {
                javax.swing.JOptionPane.showMessageDialog(null, "Selecione uma linha ou Carregue as Entidades!", "Jaw - Entidades", JOptionPane.WARNING_MESSAGE);
                return;
            } else {
                int opcao = javax.swing.JOptionPane.showConfirmDialog(null, "Tem certeza que você deseja remover esta Entidade!", "Exclusão de Entidades", JOptionPane.YES_NO_OPTION);
                if (opcao == JOptionPane.YES_OPTION) {
                    getFormAtual().getEntidades().remove(getFormAtual().getEntidadeSelecionada());
                    getFormAtual().atualizarEntidades();
                }
            }
        } catch (Exception ex) {
            jaw.Main.janelaPrincipal.log(ex.toString());
        }
    }//GEN-LAST:event_jButton3ActionPerformed
