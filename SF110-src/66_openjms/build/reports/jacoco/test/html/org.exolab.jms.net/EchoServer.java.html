package org.exolab.jms.net;

import java.rmi.AlreadyBoundException;
import java.rmi.RemoteException;

import org.exolab.jms.net.orb.ORB;
import org.exolab.jms.net.orb.ORBFactory;
import org.exolab.jms.net.proxy.Proxy;


/**
 * EchoService server
 *
 * @version     $Revision: 1.1 $ $Date: 2004/11/26 01:51:07 $
 * @author      <a href="mailto:tma@netspace.net.au">Tim Anderson</a>
 */
public class EchoServer {

    /**
     * The ORB
     */
    private ORB _orb;

    /**
     * Echo service name
     */
    private static final String ECHO_SERVICE = "echo";


    /**
     * Construct a new <code>EchoServer</code>
     *
     * @param uri the export URI
     * @throws RemoteException if the server can't be created
     * @throws AlreadyBoundException if the service can't be bound in the
     * registry
     */
    public EchoServer(String uri)
        throws AlreadyBoundException, RemoteException {
        _orb = ORBFactory.createORB(uri);
        Proxy proxy = _orb.exportObject(new EchoServiceImpl());
        _orb.getRegistry().bind(ECHO_SERVICE, proxy);
      }

    /**
     * Main line to start the server
     *
     * @param args the command line arguments
     * @throws Exception if the server can't be created
     */
    public static void main(String args[]) throws Exception {
        EchoServer server = new EchoServer(args[0]);
    }
}
