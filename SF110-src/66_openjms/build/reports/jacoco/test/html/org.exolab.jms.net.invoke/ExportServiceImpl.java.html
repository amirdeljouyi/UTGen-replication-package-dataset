package org.exolab.jms.net.invoke;

import java.rmi.RemoteException;

import org.exolab.jms.net.orb.ORB;
import org.exolab.jms.net.proxy.Proxy;


/**
 * Test class used by {@link ExportTestCase}.
 *
 * @author <a href="mailto:tma@netspace.net.au">Tim Anderson</a>
 * @version $Revision: 1.1 $ $Date: 2005/05/24 05:41:47 $
 */
public class ExportServiceImpl implements ExportService {

    /**
     * The object to export.
     */
    private final Object _object;

    /**
     * The ORB.
     */
    private final ORB _orb;


    /**
     * Construct a new <code>ExportServiceImpl</code>.
     *
     * @param object the object to export
     * @param orb the ORB to use
     */
    public ExportServiceImpl(Object object, ORB orb) {
        _object = object;
        _orb = orb;
    }

    /**
     * Exports an object to the caller.
     *
     * @return a proxy to the exported object
     * @throws RemoteException if the object can't be exported
     */
    public Proxy exportObjectTo() throws RemoteException {
        return _orb.exportObjectTo(_object);
    }
}
