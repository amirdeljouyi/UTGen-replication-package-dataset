package org.exolab.jms.selector;


/**
 * This class provides an enumeration of the supported expression types.
 *
 * @version     $Revision: 1.1 $ $Date: 2004/11/26 01:50:45 $
 * @author      <a href="mailto:tma@netspace.net.au">Tim Anderson</a>
 * @see         Expression
 * @see         SBool
 * @see         SNumber
 * @see         SString
 */
public final class Type {

    /**
     * String type
     */
    public static final Type STRING = new Type("string");

    /**
     * Numeric type
     */
    public static final Type NUMERIC = new Type("numeric");

    /**
     * Booelan type
     */
    public static final Type BOOLEAN = new Type("boolean");

    /**
     * Undefined type
     */
    public static final Type UNDEFINED = new Type("undefined");

    /**
     * The name of the type
     */
    private String _name;


    /**
     * Construct a new <code>Type</code>
     *
     * @param name the name of the type
     */
    private Type(final String name) {
        _name = name;
    }

    /**
     * Returns a string representation of this
     *
     * @return a string representation of this
     */
    public String toString() {
        return _name;
    }

} //-- Type
