package org.exolab.jms.selector;


/**
 * This class creates SObject objects corresponding to the supplied
 * java.lang.Object objects.
 *
 * @version     $Revision: 1.1 $ $Date: 2004/11/26 01:50:44 $
 * @author      <a href="mailto:tma@netspace.net.au">Tim Anderson</a>
 * @see         SObject
 */
final class SObjectFactory {

    /**
     * Private constructor
     */
    private SObjectFactory() {
    }

    /**
     * Construct a new <code>SObject</code> corresponding to that supplied
     *
     * @param value the value
     * @return a new SObject corresponding to <code>value</code>, or
     * <code>null</code> if it is null
     */
    public static SObject create(final Object value) {
        SObject result = null;
        if (value instanceof String) {
            result = new SString((String) value);
        } else if (value instanceof Number) {
            if (value instanceof Float || value instanceof Double) {
                result = new SDouble(((Number) value).doubleValue());
            } else {
                result = new SLong(((Number) value).longValue());
            }
        } else if (value instanceof Boolean) {
            if (value.equals(Boolean.TRUE)) {
                result = SBool.TRUE;
            } else {
                result = SBool.FALSE;
            }
        }

        return result;
    }

} //-- SObjectFactory
