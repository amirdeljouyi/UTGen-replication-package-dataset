package org.exolab.jms.net.uri;

import java.rmi.RemoteException;


/**
 * Exception indicating an invalid URI.
 *
 * @version     $Revision: 1.2 $ $Date: 2005/04/26 13:16:31 $
 * @author      <a href="mailto:tma@netspace.net.au">Tim Anderson</a>
 */
public class InvalidURIException extends RemoteException {

    /**
     * Serialization ID.
     */
    private static final long serialVersionUID = 1L;


    /**
     * Construct a new <code>InvalidURIException</code> with no detail nor
     * nested exception.
     */
    protected InvalidURIException() {
    }

    /**
     * Construct a new <code>InvalidURIException</code> with a detail
     * message.
     *
     * @param detail the detail message
     */
    public InvalidURIException(String detail) {
        super(detail);
    }

    /**
     * Construct a new <code>InvalidURIException</code> with a detail
     * message and nested exception.
     *
     * @param detail the detail message
     * @param nested the root cause
     */
    public InvalidURIException(String detail, Throwable nested) {
        super(detail, nested);
    }

}
