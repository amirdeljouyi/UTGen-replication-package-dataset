package org.exolab.jms.common.threads;


/**
 * Thread factory.
 *
 * @author <a href="mailto:tma@netspace.net.au">Tim Anderson</a>
 * @version $Revision: 1.1 $ $Date: 2006/02/23 11:07:05 $
 */
public class ThreadFactory
        implements EDU.oswego.cs.dl.util.concurrent.ThreadFactory {

    /**
     * The thread group which threads will belong to. May be <code>null</code>
     */
    private final ThreadGroup _group;

    /**
     * The name to prefix each thread name with
     */
    private final String _name;

    /**
     * Determines if threads will be daemon threads
     */
    private final boolean _daemon;

    /**
     * Seed value for thread names
     */
    private volatile int _nameSeed = 0;


    /**
     * Construct a new <code>ThreadFactory</code>
     *
     * @param group  the thread group. May be <code>null</code>
     * @param name   the name to prefix each thread name with
     * @param daemon if <code>true</code> all threads will be daemon threads
     */
    public ThreadFactory(ThreadGroup group, String name, boolean daemon) {
        _group = group;
        _name = name;
        _daemon = daemon;
    }

    /**
     * Create a new thread that will run the given command when started
     *
     * @param command the command to run. May be <code>null</code>
     * @return a new thread
     */
    public Thread newThread(Runnable command) {
        String name = null;
        if (_name != null) {
            int suffix = ++_nameSeed;
            name = _name + suffix;
        }
        Thread thread = new Thread(_group, command, name);
        thread.setDaemon(_daemon);
        return thread;
    }
}
