package org.exolab.jms.client;

import javax.jms.QueueReceiver;
import javax.jms.Queue;


/**
 * Client implementation of the <code>javax.jms.QueueReceiver</code> interface.
 *
 * @version     $Revision: 1.2 $ $Date: 2005/03/18 03:36:37 $
 * @author      <a href="mailto:jima@comware.com.au">Jim Alateras</a>
 * @author      <a href="mailto:tma@netspace.net.au">Tim Anderson</a>
 */
class JmsQueueReceiver
    extends JmsMessageConsumer
    implements QueueReceiver {

    /**
     * Construct a new <code>JmsQueueReceiver</code>.
     *
     * @param session the session responsible for the consumer
     * @param consumerId the consumer identifier
     * @param queue the queue to access
     * @param selector the message selector. May be <code>null</code>
     */
    public JmsQueueReceiver(JmsSession session, long consumerId, Queue queue,
                            String selector) {
        super(session, consumerId, queue, selector);
    }

    /**
     * Returns the queue associated with this receiver.
     *
     * @return the queue associated with this receiver
     */
    public Queue getQueue() {
        return (Queue) getDestination();
    }

}
