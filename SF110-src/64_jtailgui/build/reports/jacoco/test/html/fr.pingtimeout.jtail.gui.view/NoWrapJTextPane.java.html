public class NoWrapJTextPane extends JTextPane {
    public boolean getScrollableTracksViewportWidth() {
        Component parent = getParent();
        ComponentUI ui = getUI();

        return parent == null ||
                (ui.getPreferredSize(this).width <= parent.getSize().width);
    }
}
