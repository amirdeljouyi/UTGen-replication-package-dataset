    private void jButton2ActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_jButton2ActionPerformed
        if(jPasswordField1.getText()!=null && !jPasswordField1.getText().equals("") && jPasswordField2.getText()!=null && !jPasswordField2.getText().equals("") && jPasswordField2.getText().equals(jPasswordField3.getText())){
            if(jPasswordField2.getText().length()<6){
                JOptionPane.showMessageDialog(this,"Password Length Must Not Less Than 6 Char(s)!","Change Password",JOptionPane.ERROR_MESSAGE);
            }else {
                String result=adminApp.changePassword(jPasswordField1.getText(),jPasswordField2.getText());
                JOptionPane.showMessageDialog(this,result);
            }
        }else{
            JOptionPane.showMessageDialog(this,"Password Mismatch!","Change Password",JOptionPane.ERROR_MESSAGE);
        }
    }//GEN-LAST:event_jButton2ActionPerformed
